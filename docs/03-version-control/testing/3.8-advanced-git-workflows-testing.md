## 🎯 테스트 시나리오 1: Pre-commit Hooks 시스템 검증

### 1.1 Pre-commit 설치 및 설정 확인

```bash
echo "🔗 Pre-commit Hooks 시스템 검증..."

# Pre-commit 설치 확인
if command -v pre-commit &> /dev/null; then
    echo "✅ Pre-commit 설치됨: $(pre-commit --version)"
else
    echo "❌ Pre-commit 설치되지 않음"
    pip install pre-commit
fi

# .pre-commit-config.yaml 파일 확인
if [ -f ".pre-commit-config.yaml" ]; then
    echo "✅ .pre-commit-config.yaml 파일 존재"
    
    # YAML 유효성 검사
    python -c "
import yaml
try:
    with open('.pre-commit-config.yaml', 'r') as f:
        config = yaml.safe_load(f)
    print('✅ pre-commit 설정 YAML 형식 유효')
    
    repos = config.get('repos', [])
    print(f'설정된 저장소 수: {len(repos)}')
    
    # 주요 훅 확인
    hooks = []
    for repo in repos:
        hooks.extend([hook['id'] for hook in repo.get('hooks', [])])
    
    essential_hooks = ['black', 'isort', 'flake8', 'trailing-whitespace', 'end-of-file-fixer']
    for hook in essential_hooks:
        if hook in hooks:
            print(f'  ✅ {hook} 훅 설정됨')
        else:
            print(f'  ⚠️ {hook} 훅 누락')
            
except Exception as e:
    print(f'❌ pre-commit 설정 오류: {e}')
"
else
    echo "❌ .pre-commit-config.yaml 파일 없음"
fi
```

### 1.2 Pre-commit 설치 및 테스트

```bash
echo "🧪 Pre-commit 설치 및 테스트..."

# Pre-commit 훅 설치
echo "1. Pre-commit 훅 설치:"
pre-commit install
if [ $? -eq 0 ]; then
    echo "✅ Pre-commit 훅 설치 성공"
else
    echo "❌ Pre-commit 훅 설치 실패"
fi

# Commit-msg 훅도 설치 (있는 경우)
pre-commit install --hook-type commit-msg 2>/dev/null && echo "✅ Commit-msg 훅 설치됨"

echo ""
echo "2. 모든 파일에 대한 Pre-commit 실행:"
pre-commit run --all-files
pre_commit_result=$?

if [ $pre_commit_result -eq 0 ]; then
    echo "✅ 모든 pre-commit 검사 통과"
else
    echo "⚠️ Pre-commit에서 일부 이슈 발견 (정상적인 동작)"
fi

echo ""
echo "3. 특정 훅 개별 테스트:"
hooks_to_test=("black" "isort" "flake8" "trailing-whitespace")

for hook in "${hooks_to_test[@]}"; do
    echo "  $hook 훅 테스트:"
    pre-commit run "$hook" --all-files 2>/dev/null
    if [ $? -eq 0 ]; then
        echo "    ✅ $hook 통과"
    else
        echo "    ⚠️ $hook 이슈 발견"
    fi
done
```

### 1.3 커스텀 Git Hook 스크립트 테스트

```bash
echo "🔧 커스텀 Git Hook 스크립트 테스트..."

# 커스텀 훅 스크립트들 확인
custom_hooks=(
    "scripts/hooks/validate_data_files.py"
    "scripts/hooks/validate_config_files.py"
)

for hook in "${custom_hooks[@]}"; do
    if [ -f "$hook" ]; then
        echo "✅ $hook 존재"
        
        # 실행 권한 확인
        if [ -x "$hook" ]; then
            echo "  └─ 실행 권한 있음"
        else
            echo "  └─ 실행 권한 부여"
            chmod +x "$hook"
        fi
        
        # 스크립트 구문 검증
        python -m py_compile "$hook" 2>/dev/null
        if [ $? -eq 0 ]; then
            echo "  └─ Python 구문 유효"
        else
            echo "  └─ ❌ Python 구문 오류"
        fi
    else
        echo "❌ $hook 없음"
    fi
done

echo ""
echo "테스트 파일 생성하여 커스텀 훅 테스트:"

# 테스트용 데이터 파일 생성
mkdir -p test_git_hooks
echo '{"test": "data", "valid": true}' > test_git_hooks/valid.json
echo 'invalid json {' > test_git_hooks/invalid.json

# 데이터 파일 검증 훅 테스트
if [ -f "scripts/hooks/validate_data_files.py" ]; then
    echo "데이터 파일 검증 훅 테스트:"
    python scripts/hooks/validate_data_files.py test_git_hooks/valid.json
    echo "  └─ 유효한 파일 테스트 완료"
    
    python scripts/hooks/validate_data_files.py test_git_hooks/invalid.json
    echo "  └─ 무효한 파일 테스트 완료"
fi

# 테스트 파일 정리
rm -rf test_git_hooks/
```

## 🎯 테스트 시나리오 3: Git LFS 기능 테스트

### 3.1 Git LFS 설치 및 설정 확인

```bash
echo "📦 Git LFS 기능 테스트..."

# Git LFS 설치 확인
echo "1. Git LFS 설치 확인:"
if command -v git-lfs &> /dev/null; then
    echo "✅ Git LFS 설치됨: $(git lfs version)"
    
    # Git LFS 초기화 확인
    if git lfs env >/dev/null 2>&1; then
        echo "✅ Git LFS 초기화됨"
    else
        echo "Git LFS 초기화..."
        git lfs install
    fi
else
    echo "❌ Git LFS 설치되지 않음"
    echo "Git LFS 설치 중..."
    
    # Ubuntu에서 Git LFS 설치
    curl -s https://packagecloud.io/install/repositories/github/git-lfs/script.deb.sh | sudo bash
    sudo apt-get install git-lfs
    git lfs install
fi

echo ""
echo "2. .gitattributes 파일 확인:"
if [ -f ".gitattributes" ]; then
    echo "✅ .gitattributes 파일 존재"
    
    # LFS 추적 패턴 확인
    lfs_patterns=("*.pkl" "*.h5" "*.parquet" "*.zip" "*.tar.gz")
    
    for pattern in "${lfs_patterns[@]}"; do
        if grep -q "$pattern.*filter=lfs" .gitattributes; then
            echo "  ✅ $pattern LFS 추적 설정됨"
        else
            echo "  ⚠️ $pattern LFS 추적 권장"
        fi
    done
else
    echo "❌ .gitattributes 파일 없음"
fi
```

### 3.2 Git LFS 관리 스크립트 테스트

```bash
echo "🔧 Git LFS 관리 스크립트 테스트..."

if [ -f "scripts/git_lfs_manager.py" ]; then
    echo "✅ Git LFS 관리 스크립트 존재"
    
    echo ""
    echo "1. LFS 상태 확인:"
    python scripts/git_lfs_manager.py status
    
    echo ""
    echo "2. 대용량 파일 후보 검색:"
    python scripts/git_lfs_manager.py suggest --size 1.0
    
    echo ""
    echo "3. 테스트용 대용량 파일 생성 및 LFS 테스트:"
    
    # 테스트용 대용량 파일 생성 (10MB)
    mkdir -p test_lfs
    dd if=/dev/zero of=test_lfs/large_file.dat bs=1M count=10 2>/dev/null
    
    if [ -f "test_lfs/large_file.dat" ]; then
        echo "✅ 테스트용 대용량 파일 생성됨 (10MB)"
        
        # Git LFS로 추적 설정
        git lfs track "test_lfs/*.dat"
        git add .gitattributes
        git add test_lfs/large_file.dat
        
        # LFS 파일 확인
        if git lfs ls-files | grep -q "large_file.dat"; then
            echo "✅ 파일이 LFS로 추적됨"
        else
            echo "❌ LFS 추적 실패"
        fi
        
        # 테스트 파일 정리
        git reset HEAD test_lfs/large_file.dat .gitattributes 2>/dev/null
        rm -rf test_lfs/
    else
        echo "❌ 테스트 파일 생성 실패"
    fi
    
else
    echo "❌ Git LFS 관리 스크립트 없음"
fi
```

## 🎯 테스트 시나리오 6: 성능 및 확장성 테스트

### 6.1 Git 성능 최적화 설정 테스트

```bash
echo "⚡ Git 성능 최적화 설정 테스트..."

echo "1. 현재 Git 성능 관련 설정 확인:"
performance_configs=(
    "core.preloadindex"
    "core.fscache"
    "gc.auto"
    "pack.threads"
    "pack.deltaCacheSize"
    "pack.windowMemory"
    "core.autocrlf"
    "core.filemode"
)

for config in "${performance_configs[@]}"; do
    value=$(git config "$config" 2>/dev/null)
    if [ -n "$value" ]; then
        echo "  ✅ $config: $value"
    else
        echo "  ⚠️ $config: 설정되지 않음"
    fi
done

echo ""
echo "2. Git 저장소 크기 및 성능 분석:"
repo_size=$(du -sh .git 2>/dev/null | cut -f1)
object_count=$(find .git/objects -type f | wc -l)
pack_count=$(find .git/objects/pack -name "*.pack" 2>/dev/null | wc -l)

echo "  - 저장소 크기: $repo_size"
echo "  - 오브젝트 수: $object_count"
echo "  - 팩 파일 수: $pack_count"

# 성능 권장사항
if [ $object_count -gt 10000 ]; then
    echo "  💡 권장: git gc 실행으로 성능 개선 가능"
fi

echo ""
echo "3. Git 명령어 성능 테스트:"
# Git 상태 확인 성능 측정
start_time=$(date +%s%3N)
git status > /dev/null
end_time=$(date +%s%3N)
status_time=$((end_time - start_time))

echo "  - git status 실행 시간: ${status_time}ms"

if [ $status_time -lt 1000 ]; then
    echo "  ✅ Git 상태 확인 성능 양호"
else
    echo "  ⚠️ Git 상태 확인 성능 개선 필요"
fi
```

            movie-mlops-git-test \
            bash -c "
                echo '🔍 Docker 환경에서 Git 워크플로우 테스트'
                
                # Git 설정 확인
                git --version
                git config --list | grep -E '(user|core)' | head -5
                
                # Pre-commit 확인
                if command -v pre-commit &> /dev/null; then
                    echo '✅ Pre-commit 사용 가능'
                    pre-commit --version
                else
                    echo '⚠️ Pre-commit 없음'
                fi
                
                # Git 상태 확인
                git status --porcelain | head -5
                
                echo '✅ Docker 환경 Git 테스트 완료'
            "
        
        if [ $? -eq 0 ]; then
            echo "✅ Docker 환경 Git 워크플로우 테스트 통과"
        else
            echo "❌ Docker 환경 Git 워크플로우 테스트 실패"
        fi
        
        # 테스트용 이미지 정리
        docker rmi movie-mlops-git-test > /dev/null 2>&1
        
    else
        echo "❌ Docker 이미지 빌드 실패"
    fi
else
    echo "⚠️ Dockerfile.dev를 찾을 수 없습니다. Docker 테스트 스킵"
